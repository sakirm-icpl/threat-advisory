# Manual Fix Commands for VersionIntel OAuth
# Run these commands one by one:

# 1. Stop containers and remove volumes
docker-compose down -v

# 2. Remove old images 
docker rmi versionintel-backend:latest versionintel-frontend:latest

# 3. Check environment variables are correct
cat .env | grep POSTGRES

# 4. Rebuild containers with no cache
docker-compose build --no-cache

# 5. Start services
docker-compose up -d

# 6. Check logs for database connection
docker logs versionintel_backend_1

# 7. If still issues, check what environment variables the container sees:
docker exec versionintel_backend_1 env | grep POSTGRES
